 12class NetworkTopology(ABC):
 13    """
 14    Abstract class to represent a network topology.
 15    """
 16    @abstractmethod
 17    def neighbors_of(self, pid: Pid) -> ProcessSet:
 18        """Get the neighbors of the given process in the topology.
 19        
 20        Args:
 21            pid: The process identifier.
 22        
 23        Returns:
 24            A ProcessSet containing all neighbors of the given process.
 25        """
 26        pass
 27    
 28    @abstractmethod
 29    def processes(self) -> ProcessSet:
 30        """Return the set of all processes in the topology.
 31        
 32        Returns:
 33            A ProcessSet containing all processes in this topology.
 34        """
 35        pass
 36    
 37    def __len__(self) -> int:
 38        # default implementation; expected to be overridden in subclasses
 39        # for the sake of performance
 40        return len(self.processes())
 41
 42    def __contains__(self, pid: Pid) -> bool:
 43        # default implementation; expected to be overridden in subclasses
 44        # for the sake of performance
 45        return pid in self.processes()
 46    
 47    def __iter__(self) -> Iterator[Pid]:
 48        # default implementation; expected to be overridden in subclasses
 49        # for the sake of performance
 50        return iter(self.processes())

 79        In a complete graph, every process is connected to every other process.

 82            size: The number of processes in the topology. Must be positive.
 83        
 84        Returns:
 85            A CompleteGraph instance with sequential process IDs from 1 to size.

 87        Raises:
 88            ValueError: If size is not a positive integer.
 89        """
 90        if size <= 0:
 91            raise ValueError("Size must be a positive integer.")
 92        return cls.from_(Pid(i+1) for i in range(size))

136@dataclass(frozen=True)
137class Star(NetworkTopology):
138    _center: Pid
139    _leaves: frozenset[Pid]
140    
141    def __post_init__(self) -> None:
142        if len(self._leaves) < 1:
143            raise ValueError("A star topology must have at least one leaf.")
144        if self._center in self._leaves:
145            raise ValueError("Center cannot be a leaf.")
146        
147    def center(self) -> Pid:
148        return self._center
149    
150    def neighbors_of(self, pid: Pid) -> ProcessSet:
151        if pid == self._center:
152            return ProcessSet(self._leaves)
153        if pid in self._leaves:
154            return ProcessSet(self._center)
155        raise ValueError(f"Process {pid} not found in the star topology.")
156
157    def processes(self) -> ProcessSet:
158        return ProcessSet({self._center} | self._leaves)
159
160    def __len__(self) -> int:
161        return 1 + len(self._leaves)
162    def __contains__(self, pid: Pid) -> bool:
163        return pid == self._center or pid in self._leaves
164    def __iter__(self) -> Iterator[Pid]:
165        return iter({self._center} | self._leaves)
166        
167    @classmethod
168    def from_(cls, center: Pid, leaves: Iterable[Pid]) -> Self:
169        leaves = frozenset(leaves)
170        return cls(center, leaves)
171    
172    @classmethod
173    def of_size(cls, size: int) -> Self:
174        """
175        Create a star topology with the given size.
176        """
177        if size <= 1:
178            raise ValueError("Size must be greater than 1.")
179        center = Pid(1)
180        leaves = (Pid(i+2) for i in range(size - 1))
181        return cls.from_(center, leaves)

185class Arbitrary(NetworkTopology):

187    _processes: ProcessSet = field(init=False)
188    
189    def __post_init__(self) -> None:
190        object.__setattr__(self, '_processes', ProcessSet(self._neighbors.keys()))
191        
192    def neighbors_of(self, pid: Pid) -> ProcessSet:
193        return self._neighbors.get(pid, ProcessSet())
194    
195    def processes(self) -> ProcessSet:
196        return self._processes
197    
198    @classmethod
199    def from_(cls,
200              channels: Iterable[Pid | tuple[Pid, Pid] | Channel | tuple[int, int]],
201              directed: bool = True
202    ) -> Self:
203        neighbors: dict[Pid, ProcessSet] = {}
204        for entry in channels:
205            if isinstance(entry, Pid):
206                neighbors.setdefault(entry, ProcessSet())
207            else:
208                if isinstance(entry, Channel):
209                    s, r = entry.as_tuple()
210                else:
211                    s, r = Pid(entry[0]), Pid(entry[1])
212                neighbors.setdefault(s, ProcessSet())
213                neighbors[s] = neighbors[s] + r
214                if not directed:
215                    neighbors.setdefault(r, ProcessSet())
216                    neighbors[r] = neighbors[r] + s
217                    
218        return cls(neighbors)
